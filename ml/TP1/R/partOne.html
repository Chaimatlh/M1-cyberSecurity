dataset = read.csv('~/Desktop/chaimas2/ml/R/Data(1)/Data/Salary_Data.csv')
#setwd("Desktop/chaimas2/ml/R/Data(1)/Data")
# install.packages('caTools')
library(caTools)

# Split dataset into training dataset (ratio 2/3) and testing dataset (ratio 1/3)
set.seed(123) # seed choose randomly the values 
split = sample.split(dataset$Salary, SplitRatio = 2/3)
training_set = subset(dataset, split == TRUE)
test_set = subset(dataset, split == FALSE)


regressor = lm(formula = Salary ~ YearsExperience, data = training_set)

#Predicting the Test set results  
y_pred = predict(regressor, newdata = test_set)

dataset = read.csv('~/Desktop/chaimas2/ml/R/Data(1)/Data/50_Startups.csv')

training_set = subset(dataset, split == TRUE)
